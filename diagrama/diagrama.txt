Table news {
  id integer [primary key]
  title varchar
  description varchar
  body text [note: 'Content of the post']
  link varchar
  category varchar
  user_id integer
  status varchar
  imageURL varchar
  created_at timestamp
  updated_at timestamp
}

Table project {
  id integer [primary key]
  title varchar
  description varchar
  body text [note: 'Content of the post']
  link varchar
  category varchar
  status varchar
  imageURL varchar
  created_at timestamp
  updated_at timestamp
}


Table member {
  id integer [primary key]
  username varchar
  email varchar
  name varchar
  imageURL varchar
  link varchar
  role varchar
  created_at timestamp
  updated_at timestamp
}


Ref: news.user_id > member.id // many-to-one
