Table news {
  id integer [primary key]
  title varchar
  description varchar
  body text [note: 'Content of the post']
  link varchar
  category varchar
  user_id integer
  status varchar
  imageURL varchar
  created_at timestamp
  updated_at timestamp
}

Table project {
  id integer [primary key]
  title varchar
  description varchar
  body text [note: 'Content of the post']
  link varchar
  category varchar
  status varchar
  imageURL varchar
  created_at timestamp
  updated_at timestamp
}

//transformar id em UUID qdo em producao
CREATE TABLE member (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    name VARCHAR(100),
    image_url VARCHAR(255),
    link VARCHAR(255),
    role VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


Ref: news.user_id > member.id // many-to-one
