Table news {
  id integer [primary key]
  title varchar
  description varchar
  body text [note: 'Content of the post']
  link varchar
  category varchar
  user_id integer
  status varchar
  imageURL varchar
  created_at timestamp
  updated_at timestamp
}

Table project {
  id integer [primary key]
  title varchar
  description varchar
  body text [note: 'Content of the post']
  link varchar
  category varchar
  status varchar
  imageURL varchar
  created_at timestamp
  updated_at timestamp
}

CREATE TABLE member (
    id SERIAL PRIMARY KEY,
    username VARCHAR,
    email VARCHAR,
    name VARCHAR,
    imageURL VARCHAR,
    link VARCHAR,
    role VARCHAR,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);



Ref: news.user_id > member.id // many-to-one
